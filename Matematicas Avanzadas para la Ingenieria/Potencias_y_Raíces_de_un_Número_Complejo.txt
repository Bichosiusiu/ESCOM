\textbf{Escriba en forma polar el número complejo indicado.}

\section{Ejercicio 1}
\begin{equation}2\end{equation}
Aplicando la fórmula correspondiente para pasar un número complejo en forma rectangular a polar. Primero calculando el valor de r.
\begin{equation}r=\sqrt{2^{2}}\end{equation}
\begin{equation}r=2\end{equation}
Posteriormente, se cálcula el valor del ángulo.
\begin{equation}\theta=\arctan(\frac{0}{2})\end{equation}
\begin{equation}\theta=\arctan(0)\end{equation}
\begin{equation}\theta=0\end{equation}
Por lo tanto, se tiene que la forma polar del número complejo es:
\begin{equation}2(\cos(0)+i\sin({0}))\end{equation}

\section{Ejercicio 15}
\begin{equation}z_{1}=2(\cos(\frac{\pi}{8})+i\sin(\frac{\pi}{8}))\end{equation}
\begin{equation}z_{2}=4(\cos(\frac{3\pi}{8})+i\sin(\frac{3\pi}{8}))\end{equation}
Primero se obtiene $z_{1}z_{2}$, siguiendo la fórmula correspondiente.
\begin{equation}z_{1}z_{2}=(2*4)(\cos(\frac{\pi}{8}+\frac{3\pi}{8})+i\sin(\frac{\pi}{8}+\frac{3\pi}{8}))\end{equation}
\begin{equation}z_{1}z_{2}=8(\cos(\frac{\pi}{2})+i\sin(\frac{\pi}{2}))\end{equation}
\begin{equation}z_{1}z_{2}=8(0+i)\end{equation}
Por lo tanto, se tiene que la forma a+ib de $z_{1}z_{2}$ es:
\begin{equation}8i\end{equation}
Ahora, para obtener $\frac{z_{1}}{z_{2}}$, de igual forma se sigue la fórmula indicada.
\begin{equation}\frac{z_{1}}{z_{2}}=\frac{2}{4}(\cos(\frac{\pi}{8}-\frac{3\pi}{8})+i\sin(\frac{\pi}{8}-\frac{3\pi}{8}))\end{equation}
\begin{equation}\frac{z_{1}}{z_{2}}=\frac{1}{2}(\cos(-\frac{2\pi}{8})+i\sin(-\frac{2\pi}{8}))\end{equation}
\begin{equation}\frac{z_{1}}{z_{2}}=\frac{1}{2}(\frac{\sqrt{2}}{2}-\frac{\sqrt{2}}{2}i)\end{equation}
Por lo tanto, se tiene que la forma a+ib de $\frac{z_{1}}{z_{2}}$ es:
\begin{equation}\frac{\sqrt{2}}{4}-\frac{\sqrt{2}}{4}i\end{equation}

\section{Ejercicio 39}
\textbf{a) Si } $z_{1}=-1$ \textbf{ y } $z_{2}=5i$ \textbf{, verifique que } $Arg(z_{1}z_{2}) \neq Arg(z_{1}) + Arg(z_{2})$ \textbf{.}
\textbf{b) Si } $z_{1}=-1$ \textbf{ y } $z_{2}=-5i$ \textbf{, verifique que } $Arg(\frac{z_{1}}{z_{2}}) \neq Arg(z_{1}) - Arg(z_{2})$ \textbf{.}
Sea:
\begin{equation}(-1)(5i)=(0-0)+i(0-5)=-5i\end{equation}
\begin{equation}\arctan(z_{1})=\pi\end{equation}
\begin{equation}\arctan(z_{2})=\frac{\pi}{2}\end{equation}
Para la primer condición.
\begin{equation}\arctan(z_{1}z_{2})=\frac{3\pi}{2}\end{equation}
\begin{equation}\arctan(z_{1})+\arctan(z_{2})=\pi+\frac{\pi}{2}=\frac{3\pi}{2}\end{equation}
Por lo tanto, son iguales:
\begin{equation}\frac{3\pi}{2}=\frac{3\pi}{2}\end{equation}
Para la segunda condición.
\begin{equation}\arctan(\frac{z_{1}}{z_{2}})=\frac{\pi}{2}\end{equation}
\begin{equation}\arctan(z_{1})-\arctan(z_{2})=\pi-\frac{\pi}{2}=\frac{\pi}{2}\end{equation}
Por lo tanto, son iguales:
\begin{equation}\frac{\pi}{2}=\frac{\pi}{2}\end{equation}
